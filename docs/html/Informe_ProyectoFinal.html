1. Introducción

El presente documento constituye el informe final del proyecto de programación desarrollado para el curso COMP 2315 — Programación Estructurada, bajo la dirección del Dr. Edgardo Vargas Moya en la Universidad Interamericana de Puerto Rico, Recinto de Aguadilla. El proyecto consiste en el diseño e implementación de un Sistema de Expedientes Médicos del Dr. Rodríguez, una aplicación de consola e interfaz gráfica básica desarrollada íntegramente en el lenguaje de programación Java, que permite gestionar de forma digital los registros clínicos de los pacientes de un consultorio médico privado.

El sistema fue construido con el propósito de aplicar los conocimientos adquiridos a lo largo del semestre en la asignatura de Programación Estructurada. Entre sus funcionalidades principales se encuentran: un módulo de autenticación con control de intentos (login), operaciones completas de tipo CRUD (crear, buscar, actualizar y listar expedientes), persistencia de datos mediante archivos de texto plano en formato delimitado por barras verticales, y una robusta capa de validación de datos que emplea expresiones regulares (regex) para garantizar la integridad de la información ingresada. La interfaz de usuario está implementada mediante diálogos de JOptionPane de la biblioteca javax.swing, lo que brinda al usuario una experiencia interactiva sin necesidad de instalar entornos gráficos adicionales. La arquitectura del sistema se organiza en seis clases independientes que separan claramente las responsabilidades: Main, Paciente, Validador, ArchivoManager, ExpedienteManager y MenuUtils, reflejando principios de diseño orientado a objetos y buenas prácticas de desarrollo de software aprendidas en el curso.

 * Sección: [SECCIÓN]
 * Fecha de entrega: [FECHA DE ENTREGA]
 *
 * Descripción:
 *   Esta aplicación permite gestionar los expedientes médicos
 *   de los pacientes del Dr. Rodríguez mediante una interfaz
 *   de diálogos (JOptionPane) y consola.
 *
 * Importaciones utilizadas:
 *   - java.io.*      → manejo de archivos
 *   - java.util.*    → ArrayList, Scanner
 *   - javax.swing.*  → JOptionPane para la interfaz gráfica
 *   - java.text.*    → SimpleDateFormat para fechas
 *   - java.util.regex.* → Pattern y Matcher para validaciones
 */

import java.io.*;
import java.util.*;
import javax.swing.*;
import java.text.*;
import java.util.regex.*;

public class Main {

    public static void main(String[] args) {

        // ── Paso 1: Inicializar la carpeta de datos ──────────────────────────
        ArchivoManager.inicializarCarpeta();

        // ── Paso 2: Mostrar mensaje de bienvenida ────────────────────────────
        MenuUtils.mostrarMensajeBienvenida();

        // ── Paso 3: Realizar el proceso de login ─────────────────────────────
        boolean loginExitoso = MenuUtils.realizarLogin();

        // Si el login falla, cerrar la aplicación
        if (!loginExitoso) {
            JOptionPane.showMessageDialog(null,
                "No se pudo autenticar. El sistema se cerrará.",
                "Acceso Denegado", JOptionPane.ERROR_MESSAGE);
            System.exit(0);
        }

        // ── Paso 4: Crear el manejador de expedientes ────────────────────────
        ExpedienteManager manager = new ExpedienteManager();

        // ── Paso 5: Mostrar el menú principal en un bucle ────────────────────
        boolean ejecutando = true;

        // Bucle while principal — se mantiene activo hasta que el usuario salga
        while (ejecutando) {

            // Obtener la opción del menú principal
            int opcion = MenuUtils.mostrarMenuPrincipal();

            // Si el usuario cierra la ventana del menú
            if (opcion == -1) {
                opcion = 6; // Tratar como "Salir"
            }

            // Switch para manejar las opciones del menú principal
            switch (opcion) {

                case 1: // Crear nuevo expediente
                    manager.crearExpediente();
                    break;

                case 2: // Buscar expediente por número
                    manager.buscarPorNumero();
                    break;

                case 3: // Buscar expediente por fecha de visita
                    manager.buscarPorFechaVisita();
                    break;

                case 4: // Actualizar expediente
                    manager.actualizarExpediente();
                    break;

                case 5: // Listar todos los expedientes
                    manager.listarTodosLosExpedientes();
                    break;

                case 6: // Salir del sistema
                    // Confirmar antes de salir
                    int confirmar = JOptionPane.showConfirmDialog(null,
                        "¿Está seguro que desea salir del sistema?\n" +
                        "Todos los expedientes están guardados correctamente.",
                        "Confirmar Salida",
                        JOptionPane.YES_NO_OPTION,
                        JOptionPane.QUESTION_MESSAGE);

                    // Si el usuario confirma (opción "Sí")
                    if (confirmar == JOptionPane.YES_OPTION) {
                        ejecutando = false;
                    }
                    break;

                default:
                    // Esta opción no debería ocurrir, pero se maneja por seguridad
                    JOptionPane.showMessageDialog(null,
                        "Opción no reconocida. Por favor seleccione una opción válida.",
                        "Error", JOptionPane.ERROR_MESSAGE);
            }
        }

        // ── Paso 6: Mostrar mensaje de despedida y cerrar ────────────────────
        MenuUtils.mostrarMensajeDespedida();

        // Cerrar la aplicación limpiamente
        System.exit(0);
    }
}

import java.util.regex.Pattern;
import java.util.regex.Matcher;
import java.text.SimpleDateFormat;

public class Validador {

    // ─── Expresiones regulares (regex) ───────────────────────────────────────

    // Formato de Seguro Social: XXX-XX-XXXX
    private static final Pattern REGEX_SS =
        Pattern.compile("^\\d{3}-\\d{2}-\\d{4}$");

    // Formato de número de expediente: EXP-XXXXX (letras y números)
    private static final Pattern REGEX_EXP =
        Pattern.compile("^EXP-[A-Z0-9]{5}$");

    // Formato de fecha: MM/DD/YYYY
    private static final Pattern REGEX_FECHA =
        Pattern.compile("^(0[1-9]|1[0-2])/(0[1-9]|[12]\\d|3[01])/\\d{4}$");

    // ─── Validación de Seguro Social ─────────────────────────────────────────

    /**
     * Valida que el seguro social tenga formato XXX-XX-XXXX
     */
    public static boolean validarSeguroSocial(String ss) {
        if (ss == null || ss.trim().isEmpty()) return false;
        Matcher m = REGEX_SS.matcher(ss.trim());
        return m.matches();
    }

    // ─── Validación de número de expediente ──────────────────────────────────

    /**
     * Valida que el número de expediente tenga formato EXP-XXXXX
     */
    public static boolean validarNumeroExpediente(String exp) {
        if (exp == null || exp.trim().isEmpty()) return false;
        Matcher m = REGEX_EXP.matcher(exp.trim().toUpperCase());
        return m.matches();
    }

    // ─── Validación de fecha ─────────────────────────────────────────────────

    /**
     * Valida que la fecha tenga formato MM/DD/YYYY y sea una fecha real
     */
    public static boolean validarFecha(String fecha) {
        if (fecha == null || fecha.trim().isEmpty()) return false;

        // Primero verificar el patrón básico con regex
        Matcher m = REGEX_FECHA.matcher(fecha.trim());
        if (!m.matches()) return false;

        // Luego verificar que sea una fecha real (ej: no 02/30/2024)
        try {
            SimpleDateFormat sdf = new SimpleDateFormat("MM/dd/yyyy");
            sdf.setLenient(false); // modo estricto
            sdf.parse(fecha.trim());
            return true;
        } catch (Exception e) {
            return false;
        }
    }

    // ─── Validación de sexo ──────────────────────────────────────────────────

    /**
     * Valida que el sexo sea M o F (sin importar mayúsculas)
     */
    public static boolean validarSexo(String sexo) {
        if (sexo == null || sexo.trim().isEmpty()) return false;
        String s = sexo.trim().toUpperCase();
        return s.equals("M") || s.equals("F");
    }

    // ─── Validación de texto no vacío ────────────────────────────────────────

    /**
     * Valida que un campo de texto no esté vacío
     */
    public static boolean validarNoVacio(String texto) {
        return texto != null && !texto.trim().isEmpty();
    }

    // ─── Validación de plan médico ───────────────────────────────────────────

    /**
     * Valida que la opción del plan médico esté dentro del rango permitido (1-6)
     */
    public static boolean validarOpcionPlanMedico(int opcion) {
        return opcion >= 1 && opcion <= 6;
    }

    // ─── Validación de opción numérica en rango ──────────────────────────────

    /**
     * Valida que un número entero esté dentro del rango [min, max]
     */
    public static boolean validarRango(int numero, int min, int max) {
        return numero >= min && numero <= max;
    }

    // ─── Formatear campo de nombre ───────────────────────────────────────────

    /**
     * Capitaliza la primera letra de cada palabra en un nombre
     */
    public static String capitalizarNombre(String nombre) {
        if (nombre == null || nombre.trim().isEmpty()) return "";
        String[] palabras = nombre.trim().toLowerCase().split("\\s+");
        StringBuilder sb = new StringBuilder();
        for (String palabra : palabras) {
            if (palabra.length() > 0) {
                sb.append(Character.toUpperCase(palabra.charAt(0)));
                if (palabra.length() > 1) {
                    sb.append(palabra.substring(1));
                }
                sb.append(" ");
            }
        }
        return sb.toString().trim();
    }

            bw.write(paciente.toFileString());
            bw.newLine();
            return true;

            System.err.println("Error al guardar expediente: " + e.getMessage());
            return false;
        }

    /**
     * Lee todos los expedientes del archivo principal y los retorna como ArrayList.
     *
     * @return Lista de pacientes leídos del archivo
     */
    public static ArrayList<Paciente> cargarTodosLosExpedientes() {
        ArrayList<Paciente> lista = new ArrayList<>();
        File archivo = new File(ARCHIVO_PRINCIPAL);

        // Si el archivo no existe, retornar lista vacía
        if (!archivo.exists()) return lista;

        try (BufferedReader br = new BufferedReader(new FileReader(archivo))) {
            String linea;

            // Leer línea por línea con un bucle while
            while ((linea = br.readLine()) != null) {
                linea = linea.trim();
                if (!linea.isEmpty()) {
                    Paciente p = Paciente.fromFileString(linea);
                    if (p != null) {
                        lista.add(p);
                    }
                }
            }

        } catch (IOException e) {
            System.err.println("Error al cargar expedientes: " + e.getMessage());
        }

        return lista;
    }

            "Actualizar Expediente", JOptionPane.PLAIN_MESSAGE);

        if (busqueda == null || busqueda.trim().isEmpty()) {
            JOptionPane.showMessageDialog(null,
                "Operación cancelada.", "Info", JOptionPane.INFORMATION_MESSAGE);
            return;
        }

        busqueda = busqueda.trim().toUpperCase();
        Paciente paciente = null;

        for (Paciente p : listaPacientes) {
            if (p.getNumeroExpediente().equalsIgnoreCase(busqueda)) {
                paciente = p;
                break;
            }
        }

        if (paciente == null) {
            JOptionPane.showMessageDialog(null,
                "No se encontró ningún expediente con el número: " + busqueda,
                "No encontrado", JOptionPane.WARNING_MESSAGE);
            return;
        }

        boolean continuar = true;
        while (continuar) {
            String[] opciones = {
                "1. Nombre / Inicial / Apellidos",
                "2. Seguro Social",
                "3. Fecha de Nacimiento",
                "4. Sexo",
                "5. Dirección",
                "6. Plan Médico",
                "7. Fecha de Visita",
                "8. Diagnóstico",
                "9. Receta",
                "10. Fecha de Siguiente Visita",
                "11. Terminar actualización"
            };

            String seleccion = (String) JOptionPane.showInputDialog(null,
                "Paciente: " + paciente.getNombreCompleto() +
                "\nExpediente: " + paciente.getNumeroExpediente() +
                "\n\nSeleccione el campo a actualizar:",
                "Submenú de Actualización",
                JOptionPane.PLAIN_MESSAGE, null, opciones, opciones[0]);

            if (seleccion == null) break;

            int opcion = Integer.parseInt(seleccion.split("\\.")[0].trim());

            switch (opcion) {
                case 1:
                    String nuevoNombre = JOptionPane.showInputDialog(null,
                        "Nuevo NOMBRE (actual: " + paciente.getNombre() + "):",
                        "Actualizar Nombre", JOptionPane.PLAIN_MESSAGE);
                    if (nuevoNombre != null && Validador.validarNoVacio(nuevoNombre))
                        paciente.setNombre(Validador.capitalizarNombre(nuevoNombre));
                    String nuevaInicial = JOptionPane.showInputDialog(null,
                        "Nueva INICIAL (actual: " + paciente.getInicial() + "):",
                        "Actualizar Inicial", JOptionPane.PLAIN_MESSAGE);
                    if (nuevaInicial != null)
                        paciente.setInicial(nuevaInicial.trim().length() > 0
                            ? nuevaInicial.trim().substring(0, 1).toUpperCase() : "");
                    String nuevosApellidos = JOptionPane.showInputDialog(null,
                        "Nuevos APELLIDOS (actual: " + paciente.getApellidos() + "):",
                        "Actualizar Apellidos", JOptionPane.PLAIN_MESSAGE);
                    if (nuevosApellidos != null && Validador.validarNoVacio(nuevosApellidos))
                        paciente.setApellidos(Validador.capitalizarNombre(nuevosApellidos));
                    break;
                case 2:
                    String nuevoSS = "";
                    while (!Validador.validarSeguroSocial(nuevoSS)) {
                        nuevoSS = JOptionPane.showInputDialog(null,
                            "Nuevo SEGURO SOCIAL (actual: " + paciente.getSeguroSocial() +
                            ")\nFormato: XXX-XX-XXXX:",
                            "Actualizar Seguro Social", JOptionPane.PLAIN_MESSAGE);
                        if (nuevoSS == null) break;
                        if (!Validador.validarSeguroSocial(nuevoSS))
                            JOptionPane.showMessageDialog(null, "Formato inválido. Use: XXX-XX-XXXX",
                                "Error", JOptionPane.ERROR_MESSAGE);
                    }
                    if (nuevoSS != null && Validador.validarSeguroSocial(nuevoSS))
                        paciente.setSeguroSocial(nuevoSS);
                    break;
                case 3:
                    String nuevaFechaNac = "";
                    while (!Validador.validarFecha(nuevaFechaNac)) {
                        nuevaFechaNac = JOptionPane.showInputDialog(null,
                            "Nueva FECHA DE NACIMIENTO (actual: " + paciente.getFechaNacimiento() +
                            ")\nFormato: MM/DD/YYYY:", "Actualizar Fecha Nacimiento", JOptionPane.PLAIN_MESSAGE);
                        if (nuevaFechaNac == null) break;
                        if (!Validador.validarFecha(nuevaFechaNac))
                            JOptionPane.showMessageDialog(null, "Fecha inválida. Use MM/DD/YYYY.",
                                "Error", JOptionPane.ERROR_MESSAGE);
                    }
                    if (nuevaFechaNac != null && Validador.validarFecha(nuevaFechaNac))
                        paciente.setFechaNacimiento(nuevaFechaNac);
                    break;
                case 4:
                    String nuevoSexo = "";
                    while (!Validador.validarSexo(nuevoSexo)) {
                        nuevoSexo = JOptionPane.showInputDialog(null,
                            "Nuevo SEXO (actual: " + paciente.getSexo() + ")\nM o F:",
                            "Actualizar Sexo", JOptionPane.PLAIN_MESSAGE);
                        if (nuevoSexo == null) break;
                        if (!Validador.validarSexo(nuevoSexo))
                            JOptionPane.showMessageDialog(null, "Solo se acepta M o F.",
                                "Error", JOptionPane.ERROR_MESSAGE);
                    }
                    if (nuevoSexo != null && Validador.validarSexo(nuevoSexo))
                        paciente.setSexo(nuevoSexo.trim().toUpperCase());
                    break;
                case 5:
                    String nuevaDireccion = JOptionPane.showInputDialog(null,
                        "Nueva DIRECCIÓN (actual: " + paciente.getDireccion() + "):",
                        "Actualizar Dirección", JOptionPane.PLAIN_MESSAGE);
                    if (nuevaDireccion != null && Validador.validarNoVacio(nuevaDireccion))
                        paciente.setDireccion(nuevaDireccion.trim());
                    break;
                case 6:
                    String nuevoPlan = seleccionarPlanMedico();
                    if (nuevoPlan != null) paciente.setPlanMedico(nuevoPlan);
                    break;
                case 7:
                    String nuevaFechaVisita = "";
                    while (!Validador.validarFecha(nuevaFechaVisita)) {
                        nuevaFechaVisita = JOptionPane.showInputDialog(null,
                            "Nueva FECHA DE VISITA (actual: " + paciente.getFechaVisita() +
                            ")\nFormato: MM/DD/YYYY:", "Actualizar Fecha de Visita", JOptionPane.PLAIN_MESSAGE);
                        if (nuevaFechaVisita == null) break;
                        if (!Validador.validarFecha(nuevaFechaVisita))
                            JOptionPane.showMessageDialog(null, "Fecha inválida. Use MM/DD/YYYY.",
                                "Error", JOptionPane.ERROR_MESSAGE);
                    }
                    if (nuevaFechaVisita != null && Validador.validarFecha(nuevaFechaVisita))
                        paciente.setFechaVisita(nuevaFechaVisita);
                    break;
                case 8:
                    String nuevoDiag = JOptionPane.showInputDialog(null,
                        "Nuevo DIAGNÓSTICO (actual: " + paciente.getDiagnostico() + "):",
                        "Actualizar Diagnóstico", JOptionPane.PLAIN_MESSAGE);
                    if (nuevoDiag != null && Validador.validarNoVacio(nuevoDiag))
                        paciente.setDiagnostico(nuevoDiag.trim());
                    break;
                case 9:
                    String nuevaReceta = JOptionPane.showInputDialog(null,
                        "Nueva RECETA (actual: " + paciente.getReceta() + "):",
                        "Actualizar Receta", JOptionPane.PLAIN_MESSAGE);
                    if (nuevaReceta != null) paciente.setReceta(nuevaReceta.trim());
                    break;
                case 10:
                    String nuevaFechaSig = "";
                    while (!Validador.validarFecha(nuevaFechaSig)) {
                        nuevaFechaSig = JOptionPane.showInputDialog(null,
                            "Nueva FECHA DE SIGUIENTE VISITA (actual: " +
                            paciente.getFechaSiguienteVisita() + ")\nFormato: MM/DD/YYYY:",
                            "Actualizar Próxima Visita", JOptionPane.PLAIN_MESSAGE);
                        if (nuevaFechaSig == null) break;
                        if (!Validador.validarFecha(nuevaFechaSig))
                            JOptionPane.showMessageDialog(null, "Fecha inválida. Use MM/DD/YYYY.",
                                "Error", JOptionPane.ERROR_MESSAGE);
                    }
                    if (nuevaFechaSig != null && Validador.validarFecha(nuevaFechaSig))
                        paciente.setFechaSiguienteVisita(nuevaFechaSig);
                    break;
                case 11:
                    continuar = false;
                    break;
                default:
                    JOptionPane.showMessageDialog(null,
                        "Opción no válida.", "Error", JOptionPane.ERROR_MESSAGE);
            }
        }

        boolean guardado = ArchivoManager.reescribirTodosLosExpedientes(listaPacientes);
        if (guardado) {
            JOptionPane.showMessageDialog(null,
                "Expediente actualizado y guardado correctamente.",
                "Actualización exitosa", JOptionPane.INFORMATION_MESSAGE);
        } else {
            JOptionPane.showMessageDialog(null,
                "Error al guardar los cambios en el archivo.",
                "Error", JOptionPane.ERROR_MESSAGE);
        }
    }

    public void listarTodosLosExpedientes() {
        listaPacientes = ArchivoManager.cargarTodosLosExpedientes();

            JOptionPane.showMessageDialog(null,
                "No hay expedientes registrados en el sistema.",
                "Lista Vacía", JOptionPane.INFORMATION_MESSAGE);
            return;

        StringBuilder sb = new StringBuilder();
        sb.append("LISTADO DE EXPEDIENTES — Total: ")
          .append(listaPacientes.size()).append("\n");
        sb.append("═".repeat(60)).append("\n");
        sb.append(String.format("%-12s %-28s %-12s%n",
            "Expediente", "Paciente", "Fecha Visita"));
        sb.append("─".repeat(60)).append("\n");

        for (Paciente p : listaPacientes) {
            sb.append(String.format("%-12s %-28s %-12s%n",
                p.getNumeroExpediente(),
                p.getNombreCompleto().length() > 27
                    ? p.getNombreCompleto().substring(0, 27) + "…"
                    : p.getNombreCompleto(),
                p.getFechaVisita()));
        }

        sb.append("═".repeat(60));

        JOptionPane.showMessageDialog(null, sb.toString(),
            "Listado de Expedientes", JOptionPane.INFORMATION_MESSAGE);
    }

     * @return true si el login fue exitoso, false si falló

    public static boolean realizarLogin() {
        int intentos = 0;

        while (intentos < MAX_INTENTOS) {
            String usuario = JOptionPane.showInputDialog(null,
                "Ingrese su NOMBRE DE USUARIO:",
                "Inicio de Sesión", JOptionPane.PLAIN_MESSAGE);

            if (usuario == null) return false;

            String contrasena = JOptionPane.showInputDialog(null,
                "Ingrese su CONTRASEÑA:",
                "Inicio de Sesión", JOptionPane.PLAIN_MESSAGE);

            if (contrasena == null) return false;

            if (validarCredenciales(usuario.trim(), contrasena.trim())) {
                JOptionPane.showMessageDialog(null,
                    "¡Bienvenido, " + usuario.trim() + "!\n" +
                    "Acceso concedido al sistema.",
                    "Login Exitoso", JOptionPane.INFORMATION_MESSAGE);
                return true;
            } else {
                intentos++;
                int restantes = MAX_INTENTOS - intentos;

                if (restantes > 0) {
                    JOptionPane.showMessageDialog(null,
                        "Usuario o contraseña incorrectos.\n" +
                        "Intentos restantes: " + restantes,
                        "Error de Autenticación", JOptionPane.ERROR_MESSAGE);
                } else {
                    JOptionPane.showMessageDialog(null,
                        "Número máximo de intentos alcanzado.\n" +
                        "El sistema se cerrará por seguridad.",
                        "Acceso Denegado", JOptionPane.ERROR_MESSAGE);
                }
            }
        }

        return false;
    }

            if (USUARIOS[i].equals(usuario) && CONTRASENAS[i].equals(contrasena)) {
                return true;
